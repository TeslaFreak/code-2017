121 /////////////////////////////////////////////////////////////////////////
122 // Minimal printf implementation. Supports:
123 // % [+] [0] [<width>] [.precision] [l] [s|c|x|X|i|d|u|b|f]
124 //
125 // 0 == Fill field with zeroes
126 // width == Minimum field width
127 // .precision == digits of fraction
128 // l == Doubleword for x|i|d|u|o|b
129 // s == string pointer
130 // S == flash string pointer
131 // c == single character
132 // x == hex number
133 // X == uppercase string character
134 // i,d == signed decimal number
135 // u == unsigned number
136 // b == binary
137 // f == floating point
138 /////////////////////////////////////////////////////////////////////////
139 int16_t printf(const char *format, ...)
140 {
141  char print_buffer[MAX_OUTPUT_SIZE + 1];
142  va_list ap;
143  int16_t retVal;
144 
145  va_start(ap, format);
146  retVal = ssprintf(print_buffer, sizeof(print_buffer), format, ap);
147  va_end(ap);
148 
149  Serial.print(print_buffer);
150 
151  return (retVal);
152 }

169 // Like sprintf, except size of the buffer is passed in so that it won't get overrun.
170 int16_t ssprintf(char *print_buffer, int16_t buffer_size, const char *format, va_list ap)
171 {
172  PARSEOPTION_TYPE options;
173  char format_char;
174  uint16_t out_size = 0;
175 
176  do
177  {
178  format_char = pgm_read_byte(format++);
179 
180  if (format_char == '%')
181  {
182  uint16_t size;
183 
184  init_option_data(&options);
185  parse_format(&format, &options);
186  size = expand_data(&ap, &options, print_buffer, buffer_size - out_size);
187  print_buffer += size;
188  out_size += size;
189  }
190  else
191  {
192  // just a normal character, move it in the buffer.
193  *print_buffer++ = format_char;
194  out_size++;
195  }
196 
197  }
198  while ((format_char != '\0') && (out_size < buffer_size));
199 
200  // return how many characters we put into the buffer.
201  return out_size;
202 }

210 {

229 // parse the format. This routine assumes that the pointer to the format is
230 // just after the "%" character. This routine will return the pointer of
231 // the format from where it left off.
232 static void parse_format(const char **format, PARSEOPTION_TYPE *options)
233 {
234  char ch;
235 
236  options->ParseType = PARSE_INCOMPLETE;
237  while (options->ParseType == PARSE_INCOMPLETE)
238  {
239  ch = pgm_read_byte((*format)++);
240  switch (ch)
241  {
242  case '\0': // ran out of format text
243  (*format)--;
244  options->ParseType = PARSE_NONSENSE;
245  break;
246 
247  case '+':
248  options->ForceSign = true;
249  break;
250 
251  case '.':
252  options->Decimal = true;
253  options->Precision = 0;
254  break;
255 
256  case '0':
257  case '1':
258  case '2':
259  case '3':
260  case '4':
261  case '5':
262  case '6':
263  case '7':
264  case '8':
265  case '9':
266  if (options->Decimal)
267  {
268  options->Precision = (options->Precision * 10) + ch - '0';
269  }
270  else
271  {
272  options->MinWidth = (options->MinWidth * 10) + ch - '0';
273  if (options->MinWidth == 0) options->ZeroFill = true;
274  }
275  break;
276 
277  case 'l':
278  options->Long = true;
279  break;
280 
281  case 'c':
282  options->ParseType = PARSE_CHARACTER;
283  break;
284 
285  case 's':
286  options->ParseType = PARSE_STRING;
287  break;
288 
289  case 'S':
290  options->ParseType = PARSE_FLASH_STRING;
291  break;
292 
293  case 'b': /* binary */
294  options->ParseType = PARSE_BINARY;
295  break;
296 
297  case 'X': /* hex */
298  options->UpperCase = true;
299  // fall through
300  case 'x':
301  options->ParseType = PARSE_HEX;
302  break;
303 
304  case 'd': /* signed decimal */
305  case 'i': /* signed decimal */
306  options->Signed = true;
307  options->ParseType = PARSE_DECIMAL;
308  break;
309 
310  case 'u': /* unsigned decimal */
311  options->ForceSign = false; // 'u' beats '+'
312  options->ParseType = PARSE_DECIMAL;
313  break;
314 
315  case 'f': /* float */
316  options->ParseType = PARSE_FLOAT;
317  break;
318 
319  case '%':
320  options->ParseType = PARSE_PERCENT;
321  break;
322 
323  default:
324  options->ParseType = PARSE_NONSENSE;
325  break;
326  }
327  }
328 
329  return;
330 }

332 // Take each parsed format and expands the result into the 'print_buffer' based on the given 'options'.
333 // The 'print_buffer' size is passed in so it not to overflow it.
334 static uint16_t expand_data(va_list *ap, PARSEOPTION_TYPE *options, char *print_buffer, int16_t buffer_size)
335 {
336  uint16_t out_size = 0;
337 
338  if (!buffer_size) return 0; // We can probably trust outer loop won't call when there's no buffer left, but just in case
339 
340  if (options->MinWidth > buffer_size) options->MinWidth = buffer_size;
341 
342  switch (options->ParseType)
343  {
344 
345  case PARSE_NONE:
346  break;
347 
348  case PARSE_PERCENT:
349  {
350  *print_buffer++ = '%';
351  out_size++;
352  }
353  break;
354 
355  case PARSE_NONSENSE:
356  // write error code as a complete unit or not at all
357  if (buffer_size > 4)
358  {
359  *print_buffer++ = '%';
360  *print_buffer++ = '?';
361  *print_buffer++ = '?';
362  *print_buffer++ = '?';
363  out_size += 4;
364  }
365  break;
366 
367  case PARSE_BINARY:
368  {
369  uint8_t temp_num[BITS_PER_BYTE * sizeof(uint32_t)];
370  uint8_t *temp_num_ptr = temp_num;
371  int16_t temp_num_size = 0;
372 
373  // Get value from argument list and convert it to digits
374  // Decode the number to temporary storage as we need to write Most Significant Digit first
375  // to the output buffer and we don't know how much to pad yet for MinWidth.
376  if (options->Long == true)
377  {
378  uint32_t value = va_arg(*ap, uint32_t);
379  do
380  {
381  *temp_num_ptr++ = (value & 1);
382  value >>= 1;
383  temp_num_size++;
384  }
385  while (value != 0);
386  }
387  else
388  {
389  uint16_t value = va_arg(*ap, uint16_t);
390  do
391  {
392  *temp_num_ptr++ = (value & 1);
393  value >>= 1;
394  temp_num_size++;
395  }
396  while (value != 0);
397  }
398 
399  // Pad and zero fill if the user wants it.
400  {
401  // calculate what's left for leading zero's.
402  int16_t num_zeros = options->MinWidth - (temp_num_size + out_size);
403  while ((num_zeros-- > 0) && (out_size < buffer_size))
404  {
405  *print_buffer++ = (options->ZeroFill == true) ? '0' : ' ';
406  out_size++;
407  }
408  }
409 
410  // Write the converted number.
411  while ((temp_num_size-- > 0) && (out_size < buffer_size))
412  {
413  // Write the digit converted to ascii
414  *print_buffer++ = *(--temp_num_ptr) ? '1' : '0';
415  out_size++;
416  }
417  }
418  break;
419 
420  case PARSE_HEX:
421  {
422  uint8_t temp_num[NIBBLES_PER_BYTE * sizeof(uint32_t)];
423  uint8_t *temp_num_ptr = temp_num;
424  int16_t temp_num_size = 0;
425 
426  // Get value from argument list and convert it to digits
427  // Decode the number to temporary storage as we need to write Most Significant Digit first
428  // to the output buffer and we don't know how much to pad yet for MinWidth.
429  if (options->Long == true)
430  {
431  uint32_t value = va_arg(*ap, uint32_t);
432  do
433  {
434  *temp_num_ptr++ = (value & MASK(BITS_PER_NIBBLE,0));
435  value >>= BITS_PER_NIBBLE;
436  temp_num_size++;
437  }
438  while (value != 0);
439  }
440  else
441  {
442  uint16_t value = va_arg(*ap, uint16_t);
443  do
444  {
445  *temp_num_ptr++ = (value & MASK(BITS_PER_NIBBLE,0));
446  value >>= BITS_PER_NIBBLE;
447  temp_num_size++;
448  }
449  while (value != 0);
450  }
451 
452  // Pad and zero fill if the user wants it.
453  {
454  // calculate what's left for leading zero's.
455  int16_t num_zeros = options->MinWidth - (temp_num_size + out_size);
456  while ((num_zeros-- > 0) && (out_size < buffer_size))
457  {
458  *print_buffer++ = (options->ZeroFill == true) ? '0' : ' ';
459  out_size++;
460  }
461  }
462 
463  // Write the converted number.
464  while ((temp_num_size-- > 0) && (out_size < buffer_size))
465  {
466  // Write the digit converted to ascii
467  *print_buffer++ = hex_digit(*(--temp_num_ptr), options->UpperCase);
468  out_size++;
469  }
470  }
471  break;
472 
473  case PARSE_DECIMAL:
474  {
475  uint8_t temp_num[10];
476  uint8_t *temp_num_ptr = &temp_num[0];
477  int16_t temp_num_size = 0;

481  // Decode the number to temporary storage as we need to write Most Significant Digit first
482  // to the output buffer and we don't know how much to pad yet for MinWidth.
483  if (options->Long == true)

497  }
498  }
